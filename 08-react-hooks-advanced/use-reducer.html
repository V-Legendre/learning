08 — useReducer

Managing complex state transitions with a reducer function. When useState gets messy, useReducer brings order.

When to Use useReducer

useState is great for simple, independent values. useReducer is for when state updates get complex:

  • Next state depends on the previous state
  • Multiple related values change together
  • Complex state transitions with multiple conditions
  • You want a centralized place for all state logic
Python analogy: useReducer is like a finite state machine or the command/event pattern. If you've used libraries like transitions (Python FSM), or the reduce pattern in functional programming (functools.reduce), the concept is identical: (current_state, action) -> new_state.

The Pattern

const [state, dispatch] = useReducer(reducer, initialState)
1
Component dispatches action
dispatch({ type: "increment" })
->
2
Reducer receives action
reducer(currentState, action)
->
3
Returns new state
Pure function, no side effects
->
4
Component re-renders
With the new state value

Basic Example: Counter

useState version
const Counter = () => {
  const [count, setCount] = useState(0)

  return (
    <div>
      <span>{count}</span>
      <button onClick={() =>
        setCount(c => c + 1)}>
        +
      </button>
      <button onClick={() =>
        setCount(c => c - 1)}>
        -
      </button>
      <button onClick={() =>
        setCount(0)}>
        Reset
      </button>
    </div>
  )
}
useReducer version
type Action =
  | { type: "increment" }
  | { type: "decrement" }
  | { type: "reset" }

const reducer = (state: number, action: Action) => {
  switch (action.type) {
    case "increment": return state + 1
    case "decrement": return state - 1
    case "reset": return 0
  }
}

const Counter = () => {
  const [count, dispatch] = useReducer(reducer, 0)

  return (
    <div>
      <span>{count}</span>
      <button onClick={() =>
        dispatch({ type: "increment" })}>+</button>
      <button onClick={() =>
        dispatch({ type: "decrement" })}>-</button>
      <button onClick={() =>
        dispatch({ type: "reset" })}>Reset</button>
    </div>
  )
}
For this simple case, useState is fine. The reducer pattern pays off when state logic is more complex. Let's see real examples.

Discriminated Union Actions

Type your actions as a discriminated union. TypeScript will narrow the payload type inside each case block. This is one of the strongest type-safety patterns in React.

Properly typed actions
// Each action has a unique `type` and its own payload shape
type Action =
  | { type: "set_name"; payload: string }
  | { type: "set_age"; payload: number }
  | { type: "toggle_active" }                     // no payload needed
  | { type: "set_role"; payload: "admin" | "user" }

// Inside the reducer switch:
case "set_name":
  // TypeScript knows: action.payload is string
  return { ...state, name: action.payload }

case "set_age":
  // TypeScript knows: action.payload is number
  return { ...state, age: action.payload }

case "toggle_active":
  // TypeScript knows: no payload on this action
  return { ...state, isActive: !state.isActive }

Real Example: Shopping Cart

Shopping cart reducer — multiple related state transitions
type CartItem = {
  id: string
  name: string
  price: number
  quantity: number
}

type CartState = {
  items: CartItem[]
  total: number
}

type CartAction =
  | { type: "add_item"; payload: Omit<CartItem, "quantity"> }
  | { type: "remove_item"; payload: { id: string } }
  | { type: "update_quantity"; payload: { id: string; quantity: number } }
  | { type: "clear_cart" }

const calcTotal = (items: CartItem[]): number =>
  items.reduce((sum, item) => sum + item.price * item.quantity, 0)

const cartReducer = (state: CartState, action: CartAction): CartState => {
  switch (action.type) {
    case "add_item": {
      const existing = state.items.find(i => i.id === action.payload.id)
      const items = existing
        ? state.items.map(i =>
            i.id === action.payload.id
              ? { ...i, quantity: i.quantity + 1 }
              : i
          )
        : [...state.items, { ...action.payload, quantity: 1 }]
      return { items, total: calcTotal(items) }
    }

    case "remove_item": {
      const items = state.items.filter(i => i.id !== action.payload.id)
      return { items, total: calcTotal(items) }
    }

    case "update_quantity": {
      const items = state.items.map(i =>
        i.id === action.payload.id
          ? { ...i, quantity: action.payload.quantity }
          : i
      )
      return { items, total: calcTotal(items) }
    }

    case "clear_cart":
      return { items: [], total: 0 }
  }
}

// Usage in component
const Cart = () => {
  const [cart, dispatch] = useReducer(cartReducer, { items: [], total: 0 })

  return (
    <div>
      {cart.items.map(item => (
        <div key={item.id}>
          <span>{item.name} x {item.quantity}</span>
          <button onClick={() =>
            dispatch({ type: "remove_item", payload: { id: item.id } })
          }>Remove</button>
        </div>
      ))}
      <p>Total: ${cart.total.toFixed(2)}</p>
      <button onClick={() => dispatch({ type: "clear_cart" })}>
        Clear
      </button>
    </div>
  )
}

Real Example: Async State (Loading/Error/Data)

Generic async state reducer
type AsyncState<T> = {
  data: T | null
  isLoading: boolean
  error: string | null
}

type AsyncAction<T> =
  | { type: "loading" }
  | { type: "success"; payload: T }
  | { type: "error"; payload: string }

const asyncReducer = <T,>(
  state: AsyncState<T>,
  action: AsyncAction<T>,
): AsyncState<T> => {
  switch (action.type) {
    case "loading":
      return { data: null, isLoading: true, error: null }
    case "success":
      return { data: action.payload, isLoading: false, error: null }
    case "error":
      return { data: null, isLoading: false, error: action.payload }
  }
}

// Usage
const [state, dispatch] = useReducer(asyncReducer<User[]>, {
  data: null,
  isLoading: false,
  error: null,
})
In practice: Use TanStack Query for data fetching — it handles this pattern and much more. But this async reducer is excellent for understanding how reducers model state machines, and it's useful for non-HTTP async operations.

useState vs useReducer

CriterionuseStateuseReducer
State shape Single value (string, number, boolean) Object with multiple related fields
Transitions Simple: set to a new value Complex: depends on action type and current state
Related updates Multiple setState calls One dispatch updates everything atomically
Testability Test the component Reducer is a pure function — test it in isolation
Debugging Log individual state changes Log every action with timestamp — full audit trail
Example Toggle visibility, form field value Shopping cart, multi-step form, game state
Messy: Multiple useState for related values
// These are related but managed separately
const [isLoading, setIsLoading] = useState(false)
const [error, setError] = useState<string | null>(null)
const [data, setData] = useState<User[] | null>(null)

// Easy to create impossible states:
// isLoading=true AND error="oops"
// data=users AND isLoading=true

const fetchData = async () => {
  setIsLoading(true)
  setError(null)  // easy to forget this
  try {
    const result = await fetch("/api/users")
    setData(await result.json())
    setIsLoading(false)
  } catch (e) {
    setError(e.message)
    setIsLoading(false)
  }
}
Clean: useReducer for related state
// State transitions are centralized and atomic
const [state, dispatch] = useReducer(
  asyncReducer, initialState
)

// Impossible states are impossible.
// "loading" always clears error and data.
// "success" always clears loading and error.
// "error" always clears loading and data.

const fetchData = async () => {
  dispatch({ type: "loading" })
  try {
    const result = await fetch("/api/users")
    const data = await result.json()
    dispatch({ type: "success", payload: data })
  } catch (e) {
    dispatch({ type: "error",
      payload: e.message })
  }
}

Reducer Rules

What a reducer must do
  • Pure function: Same input = same output. No side effects (no API calls, no localStorage, no random values)
  • Return new state: Never mutate the existing state. Always return a new object: { ...state, field: newValue }
  • Handle all actions: TypeScript's discriminated union + switch ensures exhaustiveness
  • Synchronous: Reducers run during render. Async logic goes in event handlers or effects, then dispatch the result
Bad: Mutating state
case "add_item":
  // MUTATES the existing array!
  state.items.push(action.payload)
  return state  // same reference = no re-render
Good: Returning new state
case "add_item":
  // Creates new array with spread
  return {
    ...state,
    items: [...state.items, action.payload],
  }
Testing advantage: Since reducers are pure functions, you can test them without React at all. Just call reducer(someState, someAction) and assert the output. No rendering, no mocking. 
// reducer.test.ts — no React needed
test("add_item adds to cart", () => {
  const initial: CartState = { items: [], total: 0 }
  const result = cartReducer(initial, {
    type: "add_item",
    payload: { id: "1", name: "Widget", price: 9.99 },
  })

  expect(result.items).toHaveLength(1)
  expect(result.total).toBe(9.99)
})

test("clear_cart empties everything", () => {
  const withItems: CartState = {
    items: [{ id: "1", name: "Widget", price: 9.99, quantity: 2 }],
    total: 19.98,
  }
  const result = cartReducer(withItems, { type: "clear_cart" })

  expect(result.items).toEqual([])
  expect(result.total).toBe(0)
})

ScenarioUse
Toggle a booleanuseState
Single input valueuseState
Counter with increment/decrementEither (useState is fine)
Form with 5+ fields and validationuseReducer

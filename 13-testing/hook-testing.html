13 — Hook & Store Testing

Testing custom hooks, Zustand stores, and TanStack Query hooks outside of components.

renderHook — Testing Hooks in Isolation

Hooks can only run inside React components. renderHook creates a minimal wrapper component so you can test hooks directly.

import { renderHook, act } from "@testing-library/react"
import { useCounter } from "./useCounter"

describe("useCounter", () => {
  it("starts at initial value", () => {
    const { result } = renderHook(() => useCounter(10))

    expect(result.current.count).toBe(10)
  })

  it("increments the count", () => {
    const { result } = renderHook(() => useCounter(0))

    // Wrap state updates in act()
    act(() => {
      result.current.increment()
    })

    expect(result.current.count).toBe(1)
  })

  it("resets to initial value", () => {
    const { result } = renderHook(() => useCounter(5))

    act(() => {
      result.current.increment()
      result.current.increment()
      result.current.reset()
    })

    expect(result.current.count).toBe(5)
  })
})
Key concepts
  • result.current — always points to the latest return value of your hook
  • act() — wraps state updates so React processes them before you assert
  • rerender(newProps) — simulate prop changes to the hook

Understanding act()

act() tells React: "I'm about to trigger state updates — process them all before I check the result."

Without act()
// State update is queued but not processed
result.current.increment()

// This might see the OLD value!
expect(result.current.count).toBe(1)
// Warning: state update not wrapped in act()
With act()
// State update is processed synchronously
act(() => {
  result.current.increment()
})

// Now we see the NEW value
expect(result.current.count).toBe(1)
// No warnings
When you DON'T need act(): userEvent methods and findBy queries already wrap their operations in act(). You mainly need it when calling hook methods directly in renderHook tests.

Testing Async Hooks

// useDebounce.ts
const useDebounce = <T>(value: T, delay: number): T => {
  const [debounced, setDebounced] = useState(value)
  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay)
    return () => clearTimeout(timer)
  }, [value, delay])
  return debounced
}
// useDebounce.test.ts
describe("useDebounce", () => {
  beforeEach(() => {
    vi.useFakeTimers()  // Control time manually
  })

  afterEach(() => {
    vi.restoreAllTimers()
  })

  it("returns initial value immediately", () => {
    const { result } = renderHook(
      () => useDebounce("hello", 500)
    )
    expect(result.current).toBe("hello")
  })

  it("updates value after delay", () => {
    const { result, rerender } = renderHook(
      ({ value, delay }) => useDebounce(value, delay),
      { initialProps: { value: "hello", delay: 500 } }
    )

    // Change the input value
    rerender({ value: "world", delay: 500 })

    // Not yet — still debouncing
    expect(result.current).toBe("hello")

    // Fast-forward time
    act(() => {
      vi.advanceTimersByTime(500)
    })

    // Now it's updated
    expect(result.current).toBe("world")
  })
})
Fake timers let you control setTimeout/setInterval without actually waiting. vi.advanceTimersByTime(ms) fast-forwards. Python analogy: like freezegun but for timers instead of dates.

Testing Zustand Stores

Zustand stores are just functions — test them directly without rendering any React components.

// cartStore.ts
interface CartItem {
  id: string
  name: string
  price: number
  quantity: number
}

interface CartState {
  items: CartItem[]
  addItem: (item: Omit<CartItem, "quantity">) => void
  removeItem: (id: string) => void
  clearCart: () => void
  total: () => number
}

export const useCartStore = create<CartState>()((set, get) => ({
  items: [],
  addItem: (item) => set((state) => {
    const existing = state.items.find(i => i.id === item.id)
    if (existing) {
      return {
        items: state.items.map(i =>
          i.id === item.id ? { ...i, quantity: i.quantity + 1 } : i
        ),
      }
    }
    return { items: [...state.items, { ...item, quantity: 1 }] }
  }),
  removeItem: (id) => set((state) => ({
    items: state.items.filter(i => i.id !== id),
  })),
  clearCart: () => set({ items: [] }),
  total: () => get().items.reduce((sum, i) => sum + i.price * i.quantity, 0),
}))
// cartStore.test.ts
import { useCartStore } from "./cartStore"

describe("Cart Store", () => {
  // Reset store between tests
  beforeEach(() => {
    useCartStore.setState({ items: [] })
  })

  it("adds an item", () => {
    const { addItem } = useCartStore.getState()

    addItem({ id: "1", name: "Shirt", price: 29 })

    const { items } = useCartStore.getState()
    expect(items).toHaveLength(1)
    expect(items[0]).toEqual({
      id: "1", name: "Shirt", price: 29, quantity: 1,
    })
  })

  it("increments quantity for duplicate items", () => {
    const { addItem } = useCartStore.getState()

    addItem({ id: "1", name: "Shirt", price: 29 })
    addItem({ id: "1", name: "Shirt", price: 29 })

    const { items } = useCartStore.getState()
    expect(items).toHaveLength(1)
    expect(items[0].quantity).toBe(2)
  })

  it("removes an item", () => {
    const { addItem, removeItem } = useCartStore.getState()

    addItem({ id: "1", name: "Shirt", price: 29 })
    addItem({ id: "2", name: "Hat", price: 15 })
    removeItem("1")

    const { items } = useCartStore.getState()
    expect(items).toHaveLength(1)
    expect(items[0].name).toBe("Hat")
  })

  it("calculates total", () => {
    const { addItem, total } = useCartStore.getState()

    addItem({ id: "1", name: "Shirt", price: 29 })
    addItem({ id: "2", name: "Hat", price: 15 })
    addItem({ id: "1", name: "Shirt", price: 29 })  // 2 shirts

    expect(useCartStore.getState().total()).toBe(73)  // 29*2 + 15
  })

  it("clears the cart", () => {
    const { addItem, clearCart } = useCartStore.getState()

    addItem({ id: "1", name: "Shirt", price: 29 })
    clearCart()

    expect(useCartStore.getState().items).toEqual([])
  })
})
Why this is great

No render, no screen, no DOM — just pure function testing. Zustand stores are essentially state machines. Test the transitions.

Python analogy: This is like testing a Python class or module directly — call methods, check state. No HTTP layer, no framework overhead. The simplest kind of test.

Testing TanStack Query Hooks

Query hooks need a QueryClientProvider. Create a wrapper for your tests.

// test-utils.tsx
import { QueryClient, QueryClientProvider } from "@tanstack/react-query"
import { renderHook } from "@testing-library/react"

const createWrapper = () => {
  const queryClient = new QueryClient({
    defaultOptions: {
      queries: { retry: false },  // don't retry in tests
    },
  })

  return ({ children }: { children: React.ReactNode }) => (
    <QueryClientProvider client={queryClient}>
      {children}
    </QueryClientProvider>
  )
}
// useUsers.test.ts
import { renderHook, waitFor } from "@testing-library/react"
import { useUsers } from "./useUsers"

describe("useUsers", () => {
  it("fetches users", async () => {
    const { result } = renderHook(() => useUsers(), {
      wrapper: createWrapper(),
    })

    // Initially loading
    expect(result.current.isLoading).toBe(true)

    // Wait for data (MSW handles the API mock)
    await waitFor(() => {
      expect(result.current.isSuccess).toBe(true)
    })

    expect(result.current.data).toHaveLength(2)
    expect(result.current.data[0].name).toBe("Alice")
  })

  it("handles errors", async () => {
    // Override MSW handler for this test
    server.use(
      http.get("/api/users", () =>
        HttpResponse.json(null, { status: 500 })
      )
    )

    const { result } = renderHook(() => useUsers(), {
      wrapper: createWrapper(),
    })

    await waitFor(() => {
      expect(result.current.isError).toBe(true)
    })
  })
})

Mocking with vi.mock()

When you need to mock modules (not HTTP calls — use MSW for that).

// Mock a module
vi.mock("./analytics", () => ({
  trackEvent: vi.fn(),
  trackPageView: vi.fn(),
}))

// Import the mocked version
import { trackEvent } from "./analytics"

it("tracks click event", async () => {
  const user = userEvent.setup()
  render(<BuyButton />)

  await user.click(screen.getByRole("button"))

  expect(trackEvent).toHaveBeenCalledWith("purchase_clicked")
})

Mock Timers

beforeEach(() => vi.useFakeTimers())
afterEach(() => vi.restoreAllTimers())

it("auto-saves after 3 seconds", async () => {
  const onSave = vi.fn()
  renderHook(() => useAutoSave(data, onSave, 3000))

  expect(onSave).not.toHaveBeenCalled()

  act(() => vi.advanceTimersByTime(3000))

  expect(onSave).toHaveBeenCalledOnce()
})

What to Test: Decision Guide

ThingHow to testExample
Pure utility functionsDirect unit testexpect(formatDate(date)).toBe("Mar 30")
Zustand storesDirect — getState()/setState()Call actions, check state
Custom hooks (no UI)renderHookTest return values + state changes
Components (simple)render + screenCheck rendered output, interactions
Components (with API)render + MSWMock API, check loading/error/data states
Query hooksrenderHook + wrapper + MSWCheck isLoading → isSuccess flow
Form logicrender + userEventFill fields, submit, check validation
Rule of thumb: Test the simplest way possible. If you can test without React (stores, utils), do that. If you can test without mocks (render with real child components), do that. Only add complexity when needed.
